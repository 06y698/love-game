<script>

const tarot = [
{ name:"愚者", meaning:"新的开始，冒险与可能。" },
{ name:"恋人", meaning:"关系与选择。" },
{ name:"魔术师", meaning:"行动与掌控。" },
{ name:"女祭司", meaning:"直觉与隐藏信息。" },
{ name:"皇后", meaning:"成长与滋养。" },
{ name:"皇帝", meaning:"掌控与规则。" },
{ name:"战车", meaning:"前进与胜利。" },
{ name:"力量", meaning:"内心的坚定。" },
{ name:"命运之轮", meaning:"转机与变化。" },
{ name:"星星", meaning:"希望与疗愈。" }
];

function draw(n){
    let result=[];
    for(let i=0;i<n;i++){
        result.push(tarot[Math.floor(Math.random()*tarot.length)]);
    }
    return result;
}

function start(){
    const type=document.getElementById("type").value;
    const q=document.getElementById("question").value;

    if(!type){ alert("请选择类型"); return; }
    if(!q){ alert("请输入问题"); return; }

    let cards=[];
    let labels=[];

    if(type==="love"){
        cards=draw(3);
        labels=["过去","现在","未来"];
    }
    if(type==="career"){
        cards=draw(3);
        labels=["现状","挑战","建议"];
    }
    if(type==="money"){
        cards=draw(1);
        labels=["结果"];
    }
    if(type==="study"){
        cards=draw(2);
        labels=["问题","解决"];
    }
    if(type==="general"){
        cards=draw(3);
        labels=["现状","趋势","结果"];
    }

    let html=`<h3>✨ 你的问题：${q}</h3><div class="cards">`;

    cards.forEach((c,i)=>{
        html+=`
        <div class="card">
            <h4>${labels[i]}</h4>
            <strong>${c.name}</strong>
            <p>${c.meaning}</p>
        </div>`;
    });

    html+=`</div>`;

    document.getElementById("result").innerHTML=html;
}

</script>
