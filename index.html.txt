let stage = 0;

function updateBar(){
    for(let i=0;i<4;i++){
        document.getElementById("a"+(i+1)).style.width = affection[i]*2+"%";
    }
}

function show(text, options){
    document.getElementById("story").innerHTML = text;
    let html="";
    options.forEach((opt,i)=>{
        html+=`<button onclick="choose(${i})">${opt.text}</button>`;
    });
    document.getElementById("choices").innerHTML = html;
    window.currentOptions = options;
}

function choose(i){
    let opt = currentOptions[i];
    affection[opt.target]+=opt.aff;
    stage = opt.next;
    updateBar();
    next();
}
